            <h2 class="section-title">RPO（採用プロセスアウトソーシング）とは？</h2>

            <div class="max-w-3xl mx-auto text-center text-lg text-gray-700 space-y-4">

                <p>RPO（Recruitment Process Outsourcing）とは、企業が採用活動の一部または全体を外部の専門企業に委託するサービスです。日本では「採用代行」や「採用アウトソーシング」とも呼ばれ、採用戦略の立案から母集団形成、応募者対応、選考、内定者フォロー、さらには入社後の定着支援まで、幅広い業務をカバーします。</p>

                <p>RPOを活用することで、企業は採用業務の効率化、コスト削減、専門知識の活用、採用品質の向上、そして自社のコア業務へのリソース集中といったメリットを享受できます。変化の激しい現代のビジネス環境において、戦略的な人材獲得を実現するための有効な手段として注目されています。</p>

        function wrapLabel(str, maxWidth) {

            if (!str) return '';

            if (str.length <= maxWidth) {

                return str;

            }

            const words = str.split(' ');

            const lines = [];

            let currentLine = '';

            for (const word of words) {

                if (currentLine.length > 0 && (currentLine + ' ' + word).length > maxWidth) {

                    lines.push(currentLine.trim());

                    currentLine = '';

                }

                currentLine += (currentLine.length > 0 ? ' ' : '') + word;

            }

            if (currentLine.length > 0) {

                lines.push(currentLine.trim());

            }

            return lines.length > 0 ? lines : [str];

        }
